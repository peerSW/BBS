
CREATE TABLE `BanKuai` (
	`BanKuaiID` INT(11) NOT NULL,
	`count` INT(11) NOT NULL,
	`topic` VARCHAR(50) NOT NULL,
	`description` VARCHAR(50) NULL DEFAULT NULL,
	PRIMARY KEY (`BanKuaiID`) USING HASH
)
COLLATE='utf8_general_ci'
ENGINE=InnoDB
;

CREATE TABLE `HuiFu0` (
	`HuiFuID` CHAR(36) NOT NULL,
	`TieZiID` CHAR(36) NOT NULL,
	`userID` CHAR(36) NOT NULL,
	`username` CHAR(24) NOT NULL,
	`createTime` TIMESTAMP(3) NOT NULL DEFAULT CURRENT_TIMESTAMP(3) ON UPDATE CURRENT_TIMESTAMP(3),
	`content` TEXT NOT NULL,
	PRIMARY KEY (`HuiFuID`),
	INDEX `TieZiID` (`TieZiID`) USING HASH
)
COLLATE='utf8_general_ci'
ENGINE=InnoDB
;

CREATE TABLE `settings` (
	`settingID` INT(11) NOT NULL AUTO_INCREMENT,
	`settingName` VARCHAR(50) NULL DEFAULT '0',
	`settingValue` VARCHAR(50) NULL DEFAULT '0',
	PRIMARY KEY (`settingID`)
)
COLLATE='latin1_swedish_ci'
ENGINE=InnoDB
AUTO_INCREMENT=2
;

CREATE TABLE `TieZi1` (
	`TieZiID` CHAR(36) NOT NULL,
	`userID` CHAR(36) NOT NULL,
	`username` CHAR(24) NOT NULL,
	`createTime` TIMESTAMP(3) NOT NULL DEFAULT CURRENT_TIMESTAMP(3) ON UPDATE CURRENT_TIMESTAMP(3),
	`count` SMALLINT(5) NOT NULL,
	`updateTime` TIMESTAMP(3) NOT NULL DEFAULT CURRENT_TIMESTAMP(3),
	`title` TINYTEXT NOT NULL,
	`BanKuaiID` INT(11) NOT NULL,
	PRIMARY KEY (`TieZiID`)
)
COLLATE='utf8_general_ci'
ENGINE=InnoDB
;

CREATE TABLE `user` (
	`userID` CHAR(36) NOT NULL,
	`username` CHAR(24) NOT NULL,
	`password` CHAR(32) NOT NULL,
	PRIMARY KEY (`userID`) USING HASH,
	UNIQUE INDEX `username` (`username`)
)
COLLATE='utf8_general_ci'
ENGINE=InnoDB
;
